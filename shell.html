3 nkfとlatexの

if文での正規表現の使い方も覚えておこう 

#!/bin/bash

for s in `ls`
do
    if expr "$s" : ".*\.tex" >/dev/null; then
        echo $s"utf8->"$s
        nkf -w8 $s > temp-$s
        mv temp-$s $s
    fi
done

platex main.tex ; dvipdfmx main.dvi ; open main.pdf

4 rename

#!/bin/bash

for s in `ls`
do
    case $s in
        $1.pdf) mv $1.pdf $2.pdf ;;
        $1.dvi) mv $1.dvi $2.dvi ;;
        $1.tex) mv $1.tex $2.tex ;;
    esac
done
